import {
  createAddress,
  updateAddress,
  deleteAddress,
  setDefaultAddress,
} from "./action";
import { verifySession } from "@/lib/session";
import { redirect } from "next/navigation";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/dialog";
import { Checkbox } from "@/components/ui/checkbox";
import { Trash2, Edit, Home, Plus } from "lucide-react";
import prisma from "@/lib/prisma";

async function getUserAddresses(userId: string) {
  return await prisma.address.findMany({
    where: { userId },
    orderBy: [{ isDefault: "desc" }, { createdAt: "desc" }],
  });
}

export default async function AddressesPage() {
  const session = await verifySession();
  if (!session.success || !session.userId) {
    redirect("/signin");
  }

  const addresses = await getUserAddresses(session.userId as string);

  return (
    <div className="container mx-auto py-10 px-4 max-w-5xl">
      <div className="flex justify-between items-center mb-8">
        <h1 className="text-3xl font-bold">My Addresses</h1>

        <Dialog>
          <DialogTrigger asChild>
            <Button>
              <Plus className="mr-2 h-4 w-4" />
              Add New Address
            </Button>
          </DialogTrigger>
          <DialogContent className="sm:max-w-lg">
            <DialogHeader>
              <DialogTitle>Add New Address</DialogTitle>
            </DialogHeader>
            <AddressForm action={createAddress} />
          </DialogContent>
        </Dialog>
      </div>

      {addresses.length === 0 ? (
        <Card>
          <CardContent className="text-center py-12">
            <p className="text-gray-500 mb-4">
              You haven't added any addresses yet.
            </p>
            <Dialog>
              <DialogTrigger asChild>
                <Button variant="outline">
                  <Plus className="mr-2 h-4 w-4" /> Add Your First Address
                </Button>
              </DialogTrigger>
              <DialogContent>
                <DialogHeader>
                  <DialogTitle>Add New Address</DialogTitle>
                </DialogHeader>
                <AddressForm action={createAddress} />
              </DialogContent>
            </Dialog>
          </CardContent>
        </Card>
      ) : (
        <div className="grid gap-6 md:grid-cols-2 lg:grid-cols-3">
          {addresses.map((address) => (
            <Card
              key={address.id}
              className={`relative ${
                address.isDefault ? "ring-2 ring-primary" : ""
              }`}
            >
              {address.isDefault && (
                <div className="absolute -top-3 left-1/2 -translate-x-1/2 bg-primary text-primary-foreground text-xs px-3 py-1 rounded-full flex items-center gap-1">
                  <Home className="h-3 w-3" />
                  Default
                </div>
              )}

              <CardHeader>
                <CardTitle className="text-lg">
                  {address.street}, {address.city}
                </CardTitle>
              </CardHeader>
              <CardContent>
                <div className="text-sm text-gray-600 space-y-1">
                  {address.state && <p>{address.state}</p>}
                  <p>{address.postalCode}</p>
                  <p className="font-medium">{address.country}</p>
                </div>

                <div className="mt-6 flex gap-2">
                  {!address.isDefault && (
                    <form action={setDefaultAddress}>
                      <input
                        type="hidden"
                        name="addressId"
                        value={address.id}
                      />
                      <Button type="submit" variant="outline" size="sm">
                        Set as Default
                      </Button>
                    </form>
                  )}

                  <Dialog>
                    <DialogTrigger asChild>
                      <Button variant="outline" size="sm">
                        <Edit className="h-4 w-4 mr-1" />
                        Edit
                      </Button>
                    </DialogTrigger>
                    <DialogContent>
                      <DialogHeader>
                        <DialogTitle>Edit Address</DialogTitle>
                      </DialogHeader>
                      <AddressForm action={updateAddress} address={address} />
                    </DialogContent>
                  </Dialog>

                  <form
                    action={async () => {
                      "use server";
                      await deleteAddress(address.id);
                    }}
                  >
                    <Button type="submit" variant="destructive" size="sm">
                      <Trash2 className="h-4 w-4" />
                    </Button>
                  </form>
                </div>
              </CardContent>
            </Card>
          ))}
        </div>
      )}
    </div>
  );
}

// Reusable Address Form Component
async function AddressForm({
  action,
  address,
}: {
  action: any;
  address?: any;
}) {
  const isEdit = !!address;

  return (
    <form action={action} className="space-y-4">
      {isEdit && <input type="hidden" name="id" value={address.id} />}

      <div className="grid grid-cols-2 gap-4">
        <div className="col-span-2">
          <Label htmlFor="street">Street Address</Label>
          <Input
            id="street"
            name="street"
            defaultValue={address?.street}
            placeholder="123 Main St"
            required
          />
        </div>

        <div>
          <Label htmlFor="city">City</Label>
          <Input
            id="city"
            name="city"
            defaultValue={address?.city}
            placeholder="New York"
            required
          />
        </div>

        <div>
          <Label htmlFor="state">State / Province</Label>
          <Input
            id="state"
            name="state"
            defaultValue={address?.state || ""}
            placeholder="NY (optional)"
          />
        </div>

        <div>
          <Label htmlFor="postalCode">Postal Code</Label>
          <Input
            id="postalCode"
            name="postalCode"
            defaultValue={address?.postalCode}
            placeholder="10001"
            required
          />
        </div>

        <div>
          <Label htmlFor="country">Country</Label>
          <Input
            id="country"
            name="country"
            defaultValue={address?.country}
            placeholder="United States"
            required
          />
        </div>
      </div>

      <div className="flex items-center space-x-2">
        <Checkbox
          id="isDefault"
          name="isDefault"
          defaultChecked={address?.isDefault}
        />
        <Label htmlFor="isDefault" className="text-sm font-normal">
          Set as default address
        </Label>
      </div>

      <div className="flex justify-end gap-3 pt-4">
        <Button type="submit">
          {isEdit ? "Update Address" : "Save Address"}
        </Button>
      </div>
    </form>
  );
}
